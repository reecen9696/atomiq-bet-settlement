# Atomik SDK Usage Examples\n\n## Quick Start\n\n### 1. Installation\nCopy the `src/sdk/` folder to your new React project.\n\n### 2. Environment Setup\nCreate a `.env` file:\n\n`env\nVITE_API_BASE_URL=http://localhost:8080\nVITE_SOLANA_RPC_URL=https://api.devnet.solana.com\nVITE_SOLANA_NETWORK=devnet\nVITE_VAULT_PROGRAM_ID=BtZT2B1NkEGZwNT5CS326HbdbXzggiTYSUiYmSDyhTDJ\n`\n\n### 3. Basic Usage\n\n`tsx\nimport { useAtomikSDK } from './sdk/hooks';\nimport { useWallet } from '@solana/wallet-adapter-react';\n\nfunction MyApp() {\n  const { publicKey, sendTransaction, signTransaction } = useWallet();\n  \n  const { vault, betting, websocket } = useAtomikSDK({\n    userPublicKey: publicKey?.toString() || null,\n    sendTransaction,\n    signTransaction,\n  });\n\n  return (\n    <div>\n      {/* Vault Operations */}\n      <div>\n        <h2>Vault: {vault.balance?.toFixed(4)} SOL</h2>\n        <button onClick={() => vault.deposit(1)}>Deposit 1 SOL</button>\n        <button onClick={() => vault.withdraw(0.5)}>Withdraw 0.5 SOL</button>\n      </div>\n\n      {/* Betting */}\n      <div>\n        <button onClick={() => betting.placeCoinflipBet('heads', 0.1)}>Bet Heads (0.1 SOL)</button>\n        <button onClick={() => betting.placeCoinflipBet('tails', 0.1)}>Bet Tails (0.1 SOL)</button>\n      </div>\n\n      {/* Live Stats */}\n      <div>\n        {websocket.casinoStats && (\n          <p>Total Games: {websocket.casinoStats.totalGames}</p>\n        )}\n      </div>\n    </div>\n  );\n}\n`\n\n## Individual Service Usage\n\n### Vault Only\n\n`tsx\nimport { AtomikSDKFactory } from './sdk';\nimport { useVault } from './sdk/hooks';\n\nfunction VaultOnlyComponent() {\n  const vaultService = AtomikSDKFactory.createVaultOnly();\n  const { publicKey, sendTransaction, signTransaction } = useWallet();\n  \n  const vault = useVault(\n    publicKey?.toString() || null,\n    vaultService,\n    sendTransaction,\n    signTransaction\n  );\n\n  return (\n    <div>\n      <button onClick={() => vault.initialize()}>Create Vault</button>\n      <button onClick={() => vault.deposit(1)}>Deposit</button>\n    </div>\n  );\n}\n`\n\n### Betting Only\n\n`tsx\nimport { AtomikSDKFactory } from './sdk';\nimport { useBetting } from './sdk/hooks';\n\nfunction BettingOnlyComponent() {\n  const bettingService = AtomikSDKFactory.createBettingOnly();\n  const { publicKey } = useWallet();\n  \n  const betting = useBetting(\n    publicKey?.toString() || null,\n    bettingService\n  );\n\n  return (\n    <div>\n      <button onClick={() => betting.placeCoinflipBet('heads', 0.5)}>\n        Bet 0.5 SOL on Heads\n      </button>\n      \n      {betting.currentGame && (\n        <div>Game ID: {betting.currentGame.gameId}</div>\n      )}\n      \n      {betting.gameResult && (\n        <div>\n          Result: {betting.gameResult.outcome} \n          {betting.gameResult.won ? ' - You Won!' : ' - You Lost'}\n        </div>\n      )}\n    </div>\n  );\n}\n`\n\n### WebSocket Only\n\n`tsx\nimport { AtomikSDKFactory } from './sdk';\nimport { useWebSocket } from './sdk/hooks';\n\nfunction LiveStatsComponent() {\n  const wsManager = AtomikSDKFactory.createWebSocketOnly();\n  const ws = useWebSocket(wsManager);\n\n  return (\n    <div>\n      <div>Status: {ws.connected ? 'Connected' : 'Disconnected'}</div>\n      \n      {ws.casinoStats && (\n        <div>\n          <p>Total Games: {ws.casinoStats.totalGames}</p>\n          <p>Volume: {ws.casinoStats.totalVolume} SOL</p>\n          <p>Heads Wins: {ws.casinoStats.headsWins}</p>\n          <p>Tails Wins: {ws.casinoStats.tailsWins}</p>\n        </div>\n      )}\n      \n      <div>\n        <h3>Recent Wins</h3>\n        {ws.recentWins.map(win => (\n          <div key={win.gameId}>\n            {win.outcome}: {win.amount} SOL\n          </div>\n        ))}\n      </div>\n    </div>\n  );\n}\n`\n\n## Manual Service Usage (No React)\n\n`typescript\nimport { createAtomikSDK } from './sdk';\n\n// Create SDK instance\nconst sdk = createAtomikSDK();\n\n// Use services directly\nasync function manualBetting() {\n  // Place a bet\n  const result = await sdk.betting.placeCoinflipBet({\n    userPublicKey: 'your-wallet-pubkey',\n    choice: 'heads',\n    amount: 0.5,\n  });\n  \n  console.log('Game ID:', result.gameId);\n  \n  // Wait for result\n  const finalResult = await sdk.betting.waitForGameSettlement(result.gameId);\n  console.log('Final result:', finalResult);\n}\n\n// WebSocket usage\nasync function liveStats() {\n  const connections = await sdk.websocket.connectToCasinoStreams();\n  \n  connections.stats.subscribe('casino-stats', (stats) => {\n    console.log('Live stats:', stats);\n  });\n  \n  connections.wins.subscribe('recent-win', (win) => {\n    console.log('New win:', win);\n  });\n}\n`\n\n## Configuration\n\n`typescript\nimport { createAtomikSDK } from './sdk';\n\n// Custom configuration\nconst sdk = createAtomikSDK({\n  api: {\n    baseUrl: 'https://my-custom-api.com',\n    timeout: 10000,\n    retryAttempts: 5,\n  },\n  solana: {\n    rpcUrl: 'https://my-rpc.com',\n    commitment: 'finalized',\n  },\n  websocket: {\n    enabled: false, // Disable WebSocket\n  },\n});\n`\n\n## Benefits of This Approach\n\n✅ **Clean APIs**: Simple function calls instead of complex service classes \n✅ **Modular**: Import only what you need \n✅ **Type Safe**: Full TypeScript support \n✅ **React Ready**: Hooks handle state management \n✅ **Framework Agnostic**: Core services work without React \n✅ **Easy Configuration**: One config object controls everything \n✅ **Error Handling**: Built-in retry logic and user-friendly errors \n✅ **Testing Friendly**: Each service can be mocked independently"
